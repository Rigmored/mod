namespace = soldiertraits
	
#### Learning to ride from your guardian #####
character_event = {
	id = soldiertraits.1
	desc = "EVTDESCsoldiertraits.1"
	picture = GFX_evt_child_sword
	
	min_age = 6
	max_age = 16

	trigger = {
		guardian = {
			OR = {			
				trait = squire
				trait = good_rider
				trait = great_rider
				trait = horse_master
			}
		}
		NOT = { trait = squire }
		NOT = { trait = good_rider }
		NOT = { trait = great_rider }
		NOT = { trait = horse_master }
		OR = {
			culture = culture_variags
			culture = culture_eotheod
			culture = culture_rohirrim			
		}		
	}

	mean_time_to_happen = {
		months = 30

		# Mentor Effects
		modifier = {
			factor = 0.85
			OR = {
				guardian = { trait = squire }
				guardian = { trait = good_rider }
				guardian = { trait = great_rider }
				guardian = { trait = horse_master }
			}
		}

		# Personality Effects
		#Good
		modifier = {
			factor = 0.85
			OR = {
				trait = diligent
				trait = brave
			}
		}
		#Bad
		modifier = {
			factor = 1.35
			OR = {
				trait = slothful
				trait = craven
			}
		}		
		
		# Intelligence Effects
		#Good
		modifier = {
			factor = 0.85
			trait = genius
		}
		modifier = {
			factor = 0.9
			trait = quick
		}
		#Bad
		modifier = {
			factor = 1.2
			trait = slow
		}
		modifier = {
			factor = 1.35
			trait = imbecile
		}
		
		# Physical Effects
		#Good
		modifier = {
			factor = 0.85
			trait = strong
		}
		#Bad
		modifier = {
			factor = 2
			trait = weak
		}
		
		#Sexism
		modifier = {
			factor = 2.5
			is_female = yes
		}
		modifier = {
			factor = 1.2
			is_female = yes
			OR = {
				culture = culture_rohirrim
				culture = culture_eotheod
			}
		}
		
	}

	option = {
		name = "EVTOPTAsoldiertraits.1" # Stick 'em with the pointy end!
		hidden_tooltip = { guardian = { character_event = {id = soldiertraits.5} } } #Guardian Notification.
		add_trait = squire 
	}
}
