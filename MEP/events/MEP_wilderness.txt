namespace = colonisation

character_event = {
	id = colonisation.0
	desc = {
		trigger = { NOT = { mep_coloniz_quick = yes } }
		text = colonisation0descA
	}
	desc = {
		trigger = { mep_coloniz_quick = yes }
		text = colonisation0descB
	}
	#Do you want to colonize this province? It can take up to 20 years, and one of your demesne provinces will suffer tax and levy penalties while it expends its resources in the new lands.
	picture = GFX_evt_ruins
	is_triggered_only = yes

	option = {
		name = "colonisation0A" #Yes, our people need new lands to settle.
		mep_20_10_yr_colonist_effect = yes
		job_treasurer = {
			location = {
				add_province_modifier = {
					name = being_colonised
					duration = -1
				}
			}
		}
		if = {
			limit = { NOT = { mep_coloniz_cheap = yes } }
			wealth = -250
		}
		if = {
			limit = { mep_coloniz_cheap = yes }
			wealth = -125
		}
		mep_colonize_opinion_effect = yes
	}
	
	option = {
		name = "colonisation0opt2" #No, not now.
		trigger = { ai = no }
	}
} 

#Now the province has the modifier, MTTH of 20 years for it to be colonized
province_event = {
	id = colonisation.1
	desc = "colonisation1desc" #This province has been colonised, and will now start paying taxes and contributing troops to your realm. The wilderness has given way to settlements, and your people will surely prosper here.
	picture = GFX_evt_ruins
	
	trigger = {
		has_province_modifier = being_colonised
	}
	
	mean_time_to_happen = {
		months = 240
		modifier = {
			factor = 0.5
			mep_coloniz_quick = yes
		}
	}
	
	option = {
		name = OK
		remove_province_modifier = being_colonised
		owner = {
			ROOT = {
				culture = PREV
				religion = PREV
			}
		}
		if = {
			limit = { culture_group = culture_group_numenorean }
			add_province_modifier = { name = two_population_dunedain  duration = -1 }
		}
		if = {
			limit = { NOT = { culture_group = culture_group_numenorean } }
			add_province_modifier = { name = two_population_human  duration = -1 }
		}
	}
	
}

##AI wont use the decisions properly, we make them colonize via event instead - first for 'General'
character_event = {
	id = colonisation.12
	desc = "colonisation11desc" #doesnt matter, only AI sees this
	picture = GFX_harad_market
	only_playable = yes
	
	trigger = {
		war = no
		ai = yes
		OR = {
			AND = {
				NOT = { mep_coloniz_cheap = yes }
				wealth = 300
			}
			AND = {
				mep_coloniz_cheap = yes
				wealth = 175
			}
		}
		NOR = {
			culture_group = culture_group_numenorean
			mep_is_immortal_race_trigger = yes
			culture_group = culture_group_dwarves
			religion_group = religion_group_khazad
			religion_group = religion_group_wilderness
		}
		any_realm_province = {
			mep_wilderness_province_trigger = yes
			NOT = { mep_province_being_colonized_trigger = yes }
		}
		any_realm_province = {
			NOT = { culture = culture_wilderness }
			NOT = { has_province_modifier = sending_colonists }
		}
	}
	
	mean_time_to_happen = {
		months = 12
	}
	
	option = {
		name = OK
		random_realm_province = {
			limit = { mep_wilderness_province_trigger = yes }
			add_province_modifier = { name = being_colonised duration = -1 }
		}	
		if = {
			limit = { NOT = { mep_coloniz_cheap = yes } }
			wealth = -250
		}
		if = {
			limit = { mep_coloniz_cheap = yes }
			wealth = -125
		}
		mep_20_10_yr_colonist_effect = yes	
	}
	
}
	
##AI wont use the decisions properly, we make them colonize via event instead - now for numenorean culture group, as AI, they will only colonise with middlemen
character_event = {
	id = colonisation.13
	desc = "colonisation11desc" #doesnt matter, only AI sees this
	picture = GFX_harad_market
	only_playable = yes
	
	trigger = {
		war = no
		culture_group = culture_group_numenorean
		ai = yes
		
		OR = {
			# culture dunedain colonizes expensivly
			AND = {
				NOT = { mep_coloniz_cheap = yes }
				culture = culture_dunedain
				wealth = 650
			}
			# middle cost: cheap dunedian, or non-cheap non-dunedian
			AND = {
				mep_coloniz_cheap = yes
				culture = culture_dunedain
				wealth = 350
			}
			AND = {
				NOT = { mep_coloniz_cheap = yes }
				NOT = { culture = culture_dunedain }
				wealth = 300
			}
			# cheap non-dunedian colonization possible
			AND = {
				NOT = { culture = culture_dunedain }
				mep_coloniz_cheap = yes
				wealth = 175
			}
		}
		
		any_realm_province = {
			mep_wilderness_province_trigger = yes
			NOT = { mep_province_being_colonized_trigger = yes }
		}
		any_realm_province = {
			NOT = { culture = culture_wilderness }
			NOT = { has_province_modifier = sending_colonists }
