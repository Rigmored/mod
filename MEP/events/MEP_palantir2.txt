## Event to give the Palantir of Amon Sûl if not owned before 10318 ##
character_event = {
	id = palantir.0
	desc = "EVTDESCpalantir.0"
	picture = GFX_evt_palantir
	
	trigger = {
		NOT = { palantir_sul = { is_alive = yes } }
		has_landed_title = c_weather_hills
		age=1
		NOT = { year = 10318 }
		NOT = { culture = culture_wilderness }
		NOT = { religion = religion_wilderness }
		NOT = { has_global_flag = palantir_sul }
	}
	
	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = "EVTOPTApalantir.0" # I'll keep this Stone !
		add_trait = palantir_sul
		set_global_flag = palantir_sul
	}	
}

## A Character want to travel to visit the Elendil-Stone ##
character_event = {
	id = palantir.7
	desc = "EVTDESCpalantir.7"
	picture = GFX_evt_palantir
	
	is_triggered_only = yes

	option = {
		name = "EVTOPTApalantir.7" # I want to see the Undying Lands
		set_character_flag = visit_elostirion
		if = {
			limit = { 
				NOT = { trait = palantir_elostirion }
			}
			any_playable_ruler = {
				limit = { 
					trait = palantir_elostirion
					age = 1
				}
				character_event = { id = palantir.8 days = 100 }
			}
		}
		if = {
			limit = { 
				trait = palantir_elostirion
			}
			character_event = { id = palantir.9 days = 5 }
		}		
	}
	option = {
		name = "EVTOPTBpalantir.7" # I don't want to see the Tower of Avallonë !	
	}	
}

# The Lord of Minas Anor/Tirith will ask another dunedain to give him the Anor-Stone #
character_event = {
	id = palantir.14
	title = "EVTTITpalantir.14"
	desc = "EVTDESCpalantir.14"
	picture = GFX_evt_palantir
	border = GFX_event_normal_frame_intrigue
	
	only_playable = yes
	capable_only = yes
	prisoner = no
	only_men = yes
	min_age = 16
	
	trigger = {
		independent = no
		war = no
		has_landed_title = c_minas_tirith
		has_ambition = obj_claim_palantir_anor
		NOT = { trait = palantir }
		OR = {
			liege = {
				NOT = { character = ROOT }
				war = no
				trait = palantir
			}
			any_playable_ruler = {
				culture_group = culture_group_numenorean
				NOT = { character = ROOT }
				war = no
				trait = palantir			
			}
		}
		NOT = { has_character_flag = requested_anor_stone }
	}
	
	mean_time_to_happen = {
		months = 480
		modifier = {
			factor = 0.25
			trait = greedy
		}
		modifier = {
			factor = 0.5
			trait = envious
		}
		modifier = {
			factor = 0.25
			trait = ambitious
		}
		modifier = {
			factor = 10.0
			trait = content
		}
		modifier = {
			factor = 0.1
			trait = perceptive
		}
		modifier = {
			factor = 0.25
			trait = blood_of_numenor
		}
		modifier = {
			factor = 0.1
			OR = {
				has_landed_title = e_gondor
				has_landed_title = e_steward_gondor
			}
		}		
	}
	
	option = {
		name = "EVTOPTApalantir.14" # Ask him
		ai_chance = { factor = 100 }
		set_character_flag = requested_anor_stone
		any_playable_ruler = { 
			limit = {
				trait = palantir
				age=1			
			}
			character_event = { id = palantir.15 days = 4 } 
		}
	}
	
	option = {
		name = "EVTOPTBpalantir.14" # It's not that important
		ai_chance = { factor = 0 }
	}
}
