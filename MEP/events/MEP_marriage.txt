# Elven infertility after birth: I reduced the chance down to 40% (originally 60%)
character_event = {
	id = mepmarriage.10
	title = "EVTTITmepmarriage.10"
	desc = "EVTDESCmepmarriage.10"
	picture = GFX_evt_pregnancy
	
	is_triggered_only = yes	
	
	trigger = {
		is_female = yes
		trait = elf
	    NOT = { has_character_flag = elven_infertility }	
	}
		  
	option = {
		name = "EVTOPTAmepmarriage.10" #OK
        random = { 
			chance = 40
			if = { limit = { trait = fertile }
				remove_trait = fertile
			}
			add_trait = elven_infertility	
			set_character_flag = elven_infertility
	    }
	}  
}

# Simple infertility events
# hidden event that dispatches mepmarriage.14 or just adds not_fertile if conditions are correct
character_event = {
	id = mepmarriage.13
	hide_window = yes
	only_women = yes
	min_age = 45
	
	is_triggered_only = yes
	
	trigger = {
		NOT = { mep_is_immortal_race_trigger = yes }
		NOT = { religion_group = religion_group_wilderness }
		NOT = { religion_group = religion_group_khazad }
		NOT = { trait = not_fertile }
		NOT = { trait = dunedain_infertility }
		immortal = no  # just in case
		is_alive = yes
	}
	
	# option for Elven/Númenorean blood
	option = {
		trigger = {
			OR = {
				culture_group = culture_group_numenorean
				trait = elven_blood
			}
		}
		# High/Pure Blood Númenorean (this generally assumes death age corresponds to something like 85)
		if = {
			limit = {
				OR = {
					trait = blood_of_numenor
					trait = prince_numenor
					trait = blood_of_isildur # ??
					trait = blood_of_anarion # ??
				}
			}
			# Max menopause: 215 (early Númenorean )
			if = {
				limit = { age = 215 }
				character_event = { id = mepmarriage.14 }
			}
			# decline part 1
			else_if = {
				limit = {
					year = 7300
					age = 200
				}
				random = {
					chance = 90
					character_event = { id = mepmarriage.14 }
				}
			}
			# decline part 2
			else_if = {
				limit = {
					year = 7600
					age = 165
				}
				random = {
					chance = 92
					character_event = { id = mepmarriage.14 }
				}
			}
			# late 2nd age 
			else_if = {
				limit = {
					year = 7900
					age = 140
				}
				random = {
					chance = 94
					character_event = { id = mepmarriage.14 }
				}
			}
			# Into the 3rd age : 125
			else_if = {
				limit = {
					year = 9100
					age = 125
				}
				random = {
					chance = 95
					character_event = { id = mepmarriage.14 }
				}
			}
			# mid 3rd age
			else_if = {
				limit = {
					year = 9750
					age = 105
				}
				random = {
					chance = 95
					character_event = { id = mepmarriage.14 }
				}
			}
			# later 3rd age : 90 (Children and grandchildren of Aragorn are excluded, if pureblooded)
			else_if = {
				limit = {
					year = 10400
					age = 90
					NOT = { is_child_of = 2038 }
					NOT = { is_grandchild_of = 2038 }
				}
				random = {
					chance = 95
					character_event = { id = mepmarriage.14 }
				}
			}
		}
		# 'Medium Blood'
		else_if = {
			limit = { NOT = { trait = elven_blood } }
			if = {
				limit = { age = 65 }
				character_event = { id = mepmarriage.14 }
			}
		}
		# Elven Blood (and no other Númenorean traits }
		else = {
			if = {
				limit = { age = 85 }
				character_event = { id = mepmarriage.14 }
			}
		}
	}
	
	# Half-Orcs with Elven/Númenorean Blood ?
	
	# Option for Hobbits
	option = {
		trigger = {
			religion_group = religion_group_periannath
		}
		if = {
			limit = { age = 50 }
			mep_addinfertile = yes
		}
	}
	
	# Option for Other Humans
	option = {
		trigger = {
			mep_is_human_trigger = yes
			NOT = { culture_group = culture_group_numenorean }
			NOT = { trait = elven_blood }
		}
		if = {
			limit = { age = 45 }
			mep_addinfertile = yes
		}
	}
	
	# Orcs/Goblins
	option = {
		trigger = {
			mep_orc_goblin_culture = yes
		}
		if = {
			limit = { age = 45 }
			mep_addinfertile = yes
		}
	}
	# Trolls
	option = {
		trigger = {
			religion = religion_olog
		}
		if = {
			limit = { age = 45 }
			mep_addinfertile = yes
		}
	}
	
	# Misc unaccounted for crossbreeds and such ?

}
