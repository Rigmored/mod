### Balrog restless ###
character_event = {
	id = sacrificeflame.0
	title = EVTTITsacrificeflame.0
	desc = EVTDESCsacrificeflame.0
	picture = GFX_evt_worship_balrog
	
	only_rulers = yes
	capable_only = yes
	prisoner = no
	
	is_triggered_only = yes
	
	option = {
		name = EVTOPTAsacrificeflame.0 #Slaughter prisoner
		trigger = {
			any_courtier = {
				prisoner = yes
				host = { character = ROOT }
				age = 16
				OR = {
					religion_group = religion_group_eldar
					religion_group = religion_group_periannath
					religion_group = religion_group_edain
					religion_group = religion_group_hildor
				}
			}
		}
		
	    set_character_flag = flamesacrificestarted
		set_character_flag = offeredgenericprisoner
		any_courtier = {
			limit = {
				prisoner = yes
				host = { character = ROOT }
				OR = {
					religion_group = religion_group_eldar
					religion_group = religion_group_periannath
					religion_group = religion_group_edain
					religion_group = religion_group_hildor
				}
			}
			death = {
				death_reason = death_sacrificed
				killer = ROOT
			}
		}
		hidden_tooltip = {
			character_event = {
				id = sacrificeflame.1
				days = 5
				random = 5
			}
		}
	}
	
	option = {
		name = EVTOPTBsacrificeflame.0 #Slaughter dwarf
		trigger = {
			any_courtier = {
				prisoner = yes
				host = { character = ROOT }
				age = 16
				religion_group = religion_group_khazad
			}
		}
		set_character_flag = flamesacrificestarted
		set_character_flag = offereddwarfprisoner
		any_courtier = {
			limit = {
				prisoner = yes
				host = { character = ROOT }
				age = 16
				religion_group = religion_group_khazad
			}
			death = {
				death_reason = death_sacrificed
				killer = ROOT
			}
		}
		hidden_tooltip = {
			character_event = {
				id = sacrificeflame.2
				days = 5
				random = 5
			}
		}
	}
	
	option = {
		name = EVTOPTCsacrificeflame.0 #Sacrifice orks/goblins
		trigger = {
			any_courtier = {
				prisoner = yes
				host = { character = ROOT }
				age = 16
				mep_orc_goblin_culture = yes
			}
		}
		set_character_flag = flamesacrificestarted	
		set_character_flag = offeredorkprisoner
		any_courtier = {
			limit = {
