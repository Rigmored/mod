character_event = {
	id = hobbits.0
	title = "EVTTIThobbits.0"
	desc = "EVTDESChobbits.0"
	picture = "GFX_evt_hobbitsmial"
	
	capable_only = yes
	prisoner = no
	
	is_triggered_only = yes
	
	trigger = {
		has_ambition = obj_improve_diplomacy
		NOT = { has_character_flag = visit_neighbour_diplo }
		culture_group = culture_group_halfling
		capital_scope = {
			any_neighbor_province = {
				owner = { 
					same_realm = ROOT
					age = 16
					culture_group = culture_group_halfling
				}
			}
		}
	}
	
	immediate = {		
		set_character_flag = visit_neighbour_diplo
		capital_scope = {
			random_neighbor_province = {
				limit = { 
					owner = { 
						same_realm = ROOT
						age = 16
						culture_group = culture_group_halfling
					}
				}
				owner = { character_event = { id = hobbits.1 } }
			}
		}
	}
	
	option = {
		name = "EVTOPThobbits.0"
	}
}

## Will you hold a party? ##
character_event = {
	id = hobbits.6
	title = "EVTTIThobbits.6"
	desc = "EVTDESChobbits.6"
	picture = "GFX_evt_hobbitsmial"
	
	capable_only = yes
	prisoner = no
	is_triggered_only = yes
	
	trigger = {
		NOT = { has_global_flag = hobbit_party }
		culture_group = culture_group_halfling
		is_ruler = yes
		NOT = { had_character_flag = { flag=party_host days=365 } }
		wealth = 100
	}
	
	immediate = {
		set_global_flag = hobbit_party
	}
	
	option = {
		name = "EVTOPTAhobbits.6" ## Of course i will ! Family
		set_character_flag = party_host
		set_character_flag = family_party
		any_dynasty_member = {
			set_character_flag = family_party
			character_event = { id = hobbits.9 days = 1 }
		}
		character_event = { id = hobbits.10 days = 2 }
		ai_chance = {
			factor = 10
			modifier = {
				factor = 4
				trait = kind
			}
			modifier = {
				factor = 4
				NOT = { wealth = 150 }
			}
		}
	}	
	
	option = {
		name = "EVTOPTBhobbits.6" ## Of course i will ! Notable hobbits
		set_character_flag = party_host
		set_character_flag = owners_party
		any_dynasty_member = {
			set_character_flag = owners_party
			set_character_flag = invitation
			character_event = { id = hobbits.9 days = 1 }
		}
		any_playable_ruler = {
			limit = {
				culture_group = culture_group_halfling
				NOT = { dynasty = ROOT }
				NOT = { has_character_flag = invitation }
				wealth = 120
			}
			set_character_flag = owners_party
			set_character_flag = invitation
			character_event = { id = hobbits.9 days = 1 }
		}		
		character_event = { id = hobbits.10 days = 2 }	
		ai_chance = {
			factor = 10
			modifier = {
				factor = 3
				trait = gregarious
			}
			modifier = {
				factor = 4
				wealth = 200
			}
		}
	}	
	
	option = {
		name = "EVTOPTChobbits.6" ## Of course i will ! All Shire
		trigger = {
			prestige = 500
			wealth = 200
		}
		set_character_flag = party_host
		set_character_flag = shire_party
		any_dynasty_member = {
			set_character_flag = shire_party
			set_character_flag = invitation
			character_event = { id = hobbits.9 days = 1 }
		}
		any_playable_ruler = {
			limit = {
				culture_group = culture_group_halfling
				NOT = { dynasty = ROOT }
				NOT = { has_character_flag = invitation }
			}
			set_character_flag = shire_party
			set_character_flag = invitation
			character_event = { id = hobbits.9 days = 1 }
			any_dynasty_member = {
				limit = {
					NOT = { character = PREV }
					NOT = { has_character_flag = invitation }
				}
				set_character_flag = shire_party
				set_character_flag = invitation
				character_event = { id = hobbits.9 days = 1 }
			}
		}		
		character_event = { id = hobbits.10 days = 2 }
		ai_chance = {
			factor = 10
			modifier = {
				factor = 4
				trait = gregarious
			}
			modifier = {
				factor = 5
				wealth = 300
			}
		}	
	}	
	
	option = {
		name = "EVTOPTDhobbits.6" ## No, I won't.

	}
