obj_become_a_rider = {
	type = character
	allow = {
		OR = {
			is_female = no
			AND = {
				OR = {
					trait = brave
					trait = strong
					trait = tall
					trait = imposing
					trait = patient
				}
				OR = {
					culture = culture_variags
					culture = culture_eotheod
					culture = culture_rohirrim			
				}
			}
		}
		OR = {
			culture = culture_variags
			culture = culture_eotheod
			culture = culture_rohirrim			
		}		
		is_adult = yes
		martial = 5
		NOT = { trait = incapable }
		NOT = {trait = dwarf}
		NOT = {trait = weak}
		NOT = { trait = squire }
		NOT = { trait = good_rider }
		NOT = { trait = great_rider }
		NOT = { trait = horse_master }
		NOT = { has_character_flag = obj_become_a_rider }
		NOT = { has_character_flag = not_a_rider }
	}
	chance = {
		factor = 100

		modifier = {
			factor = 2
			trait = honorable
		}
		modifier = {
			factor = 0.5
			is_female = yes
		}
		modifier = {
			factor = 2.0
			trait = ambitious
		}
		modifier = {
			factor = 2.0
			trait = proud
		}
		modifier = {
			factor = 2.0
			OR = {
				trait = strong
				trait = tall
			}
		}
		modifier = {
			factor = 2.0
			trait = duelist
		}
		modifier = {
			factor = 1.5
			OR = {
				trait = tough_soldier
				trait = skilled_tactician
				trait = brilliant_strategist
			}
		}
	}
	success = {
		OR = { 
			trait = squire
			trait = good_rider
			trait = great_rider
			trait = horse_master			
		}
	}
	abort = {
		OR = {
			has_character_flag = abort_obj
			has_character_flag = not_a_rider
			trait = incapable
			age = 45
			NOT = { martial = 5 }
		}
	}
	abort_effect = {
		clr_character_flag = abort_obj
		clr_character_flag = obj_check
	}
	effect = {
		prestige = 10
		set_character_flag = obj_become_a_rider
	}
}
